% Pandoc headings: H1 = name header, H2 = major section heading.
\let\oldsection\section
\renewcommand{\section}[1]{%
  \ifcompanyblock
    \end{companyindentblock}%
    \companyblockfalse%
  \fi
  \ifheaderblock
    \end{center}%
    \headerblockfalse%
  \fi
  \oldsection{\MakeUppercase{#1}}%
  \vspace{0em}%
  \begin{center}%
  \headerblocktrue%
}

\let\oldsubsection\subsection
\renewcommand{\subsection}[1]{%
  \ifcompanyblock
    \end{companyindentblock}%
    \companyblockfalse%
  \fi
  \ifheaderblock
    \end{center}%
    \headerblockfalse%
    \vspace{0.19em}%
  \fi
  \oldsubsection{#1}\sectionrule%
}

\let\oldsubsubsection\subsubsection
\renewcommand{\subsubsection}[1]{%
  \ifcompanyblock
    \end{companyindentblock}%
    \companyblockfalse%
  \fi
  \oldsubsubsection{#1}%
  \begin{companyindentblock}%
  \companyblocktrue%
}

\AtEndDocument{%

  \fi

  \fi
